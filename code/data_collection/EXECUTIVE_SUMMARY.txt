EXAMPLE CODE FOR YOUR ANALYSIS:
================================

import pandas as pd
import numpy as np

# Load processed data
df = pd.read_csv('data/processed/data_merged_with_series.csv')

# Check available series
series_ref = pd.read_csv('data/processed/itu_series_reference.csv')
print(series_ref[series_ref['variable'] == 'fixed_broad_price'])

# Select series for regression
price_col = 'fixed_broad_price_i154_FBB_ts_PPP'  # Main specification

# Prepare panel data
panel = df[[
    'country', 'year', 'region',
    price_col,
    'gdp_per_capita',
    'population',
    'fixed_broadband_subs'
]].dropna()

# Log transformations
panel['log_price'] = np.log(panel[price_col])
panel['log_gdp'] = np.log(panel['gdp_per_capita'])

# Your fixed effects regression
from linearmodels import PanelOLS

panel = panel.set_index(['country', 'year'])
model = PanelOLS(panel['log_price'], panel[['log_gdp']], entity_effects=True)
results = model.fit()
print(results)
